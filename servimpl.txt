package com.peoplehum.competencyManagement.v1.service.impl;

import com.peoplehum.competencyManagement.common.exception.CompetencyMgmtException;
import com.peoplehum.competencyManagement.common.model.CompetencyManagementStatusCode;
import com.peoplehum.competencyManagement.common.model.SearchFilter;
import com.peoplehum.competencyManagement.v1.entity.Competency;
import com.peoplehum.competencyManagement.v1.enums.CompetencyStatus;
import com.peoplehum.competencyManagement.v1.repository.CompetencyRepository;
import com.peoplehum.competencyManagement.v1.service.CompetencyService;
import lombok.extern.slf4j.Slf4j;
import org.apache.commons.lang3.StringUtils;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.beans.factory.annotation.Qualifier;
import org.springframework.data.domain.Page;
import org.springframework.data.domain.Pageable;
import org.springframework.stereotype.Service;
import org.springframework.transaction.annotation.Transactional;

import java.util.List;
import java.util.Set;

@Service("com.peoplehum.competencyManagement.v1.service.impl.CompetencyServiceImpl")
@Slf4j
public class CompetencyServiceImpl implements CompetencyService {

  @Autowired
  @Qualifier("com.peoplehum.competencyManagement.v1.repository.CompetencyRepository")
  private CompetencyRepository competencyRepository;

  @Override
  public List<Competency> findByCustomerIdAndCompetencyGroupId(Long customerId,
      Long competencyGroupId) {
    return competencyRepository.findByCustomerIdAndCompetencyGroupId(customerId, competencyGroupId);
  }

  @Override
  public Competency save(Competency competency) {
    return competencyRepository.saveAndFlush(competency);
  }

  @Override
  public Competency findByCustomerIdAndId(Long customerId, Long id) {
    return competencyRepository.findByCustomerIdAndId(customerId, id);
  }

  @Override
  public List<Competency> findByCustomerIdAndCompetencyNameIgnoreCase(Long customerId,
      String name) {
    return competencyRepository.findByCustomerIdAndCompetencyNameIgnoreCase(customerId, name);
  }

  @Override
  public List<Competency> findByCustomerIdAndIdIn(Long customerId, List<Long> id) {
    return competencyRepository.findByCustomerIdAndIdIn(customerId, id);
  }

  @Override
  public List<Competency> findByCustomerIdAndStatusNotAndCompetencyGroupIdAndCompetencyNameIgnoreCase(
      Long customerId, CompetencyStatus status, Long competencyGroupId, String competencyName) {
    return competencyRepository.findByCustomerIdAndStatusNotAndCompetencyGroupIdAndCompetencyNameIgnoreCase(
        customerId, status, competencyGroupId, competencyName);
  }

  @Override
  public List<Competency> findByCustomerIdAndStatusNotAndCompetencyNameIgnoreCase(Long customerId,
      CompetencyStatus status, String competencyName) {
    return competencyRepository.findByCustomerIdAndStatusNotAndCompetencyNameIgnoreCase(customerId,
        status, competencyName);
  }

  @Override
  public List<Competency> findAllByCustomerId(Long customerId) {
    return competencyRepository.findAllByCustomerId(customerId);
  }

  @Override
  public Page<Competency> findAllByCustomerIdAndStatusAndCompetencyNameContainingIgnoreCase(
      Long customerId, CompetencyStatus status, String competencyName, Pageable pageable) {
    Page<Competency> returnCompetency;
    if (StringUtils.isEmpty(competencyName)) {
      returnCompetency =
          competencyRepository.findAllByCustomerIdAndStatus(customerId, status, pageable);
    } else {
      returnCompetency =
          competencyRepository.findAllByCustomerIdAndStatusAndCompetencyNameContainingIgnoreCase(
              customerId, status, competencyName, pageable);
    }
    return returnCompetency;
  }

  @Override
  public Page<Competency> findAllByCustomerId(Long customerId, SearchFilter searchFilter,
      Pageable pageable) {

    return competencyRepository.findAllByCustomerId(customerId, searchFilter.hasCompetencyIds(),
        searchFilter.getCompetencyIds(), searchFilter.hasCompetencyGroups(),
        searchFilter.getCompetencyGroupIds(), searchFilter.hasJobFunctions(),
        searchFilter.getJobFunctionIds(), searchFilter.hasPayGradeIds(),
        searchFilter.getPayGradeIds(), searchFilter.hasUpdatedOnFrom(),
        searchFilter.getUpdatedOnFrom(), searchFilter.hasUpdatedOnTo(),
        searchFilter.getUpdatedOnTo(), searchFilter.hasUpdatedBy(), searchFilter.getUpdatedBy(),
        searchFilter.hasStatuses(), searchFilter.getStatuses(), pageable);
  }

  /**
   * method to delete competency based on below params
   *
   * @param customerId
   * @param competencyId
   */
  @Transactional
  @Override
  public void inactivateCompetency(Long customerId, Long competencyId) {
    log.info("CompetencyServiceImpl: Request to mark competency as INACTIVE with customerId: {} "
        + "and competencyId: {}", customerId, competencyId);
    Integer recordCount =
        competencyRepository.updateStatusByCustomerIdAndId(customerId, competencyId);
    if (recordCount < 1) {
      log.error("Record not updated with customerId: {} and competencyId: {}", customerId,
          competencyId);
      throw new CompetencyMgmtException(CompetencyManagementStatusCode.PROCESSING_ERROR);
    }
  }

  /**
   * method to find competencies based on below params
   *
   * @param customerId
   * @param status
   *
   * @return
   */
  @Override
  public List<Competency> findAllByCustomerIdAndStatus(Long customerId, CompetencyStatus status) {
    log.info("CompetencyServiceImpl: Request to find competencies with customerId: {} "
        + "and status: {}", customerId, status);
    return competencyRepository.findAllByCustomerIdAndStatus(customerId, status);
  }

  /**
   * method to fetch count competencies by status with customerId
   *
   * @param customerId
   *
   * @return
   */
  @Override
  public Set<CompetencyRepository.StatusAndCountModelInterface> countByCustomerId(Long customerId) {
    log.info("CompetencyServiceImpl: Request to count competencies by status with customerId: {} ",
        customerId);
    return competencyRepository.countByCustomerId(customerId);
  }

  /**
   * method to fetch competencies applicable for given job function and pay-grade
   *
   * @param customerId
   * @param jobFunctionId
   * @param payGradeId
   * @param queryValue
   * @param hasQueryValue
   * @param pageable
   *
   * @return
   */
  @Override
  public Page<Competency> findAllApplicableCompetencies(Long customerId, Long jobFunctionId,
      Long payGradeId, String queryValue, Boolean hasQueryValue, Pageable pageable) {
    return competencyRepository.findAllApplicableCompetencies(customerId, jobFunctionId, payGradeId,
        queryValue, hasQueryValue, pageable);
  }
}
