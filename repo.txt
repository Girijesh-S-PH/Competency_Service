package com.peoplehum.competencyManagement.v1.repository;

import com.peoplehum.competencyManagement.v1.entity.Competency;
import com.peoplehum.competencyManagement.v1.enums.CompetencyStatus;
import org.springframework.data.domain.Page;
import org.springframework.data.domain.Pageable;
import org.springframework.data.jpa.repository.EntityGraph;
import org.springframework.data.jpa.repository.JpaRepository;
import org.springframework.data.jpa.repository.Modifying;
import org.springframework.data.jpa.repository.Query;
import org.springframework.data.repository.query.Param;
import org.springframework.stereotype.Repository;

import java.sql.Timestamp;
import java.util.List;
import java.util.Set;

import static com.peoplehum.competencyManagement.v1.constants.SqlConstants.FIND_ALL_APPLICABLE_COMPETENCIES_POSTFIX_V1;
import static com.peoplehum.competencyManagement.v1.constants.SqlConstants.FIND_ALL_COMPETENCIES_POSTFIX_V1;

@Repository("com.peoplehum.competencyManagement.v1.repository.CompetencyRepository")
public interface CompetencyRepository extends JpaRepository<Competency, Long> {

  List<Competency> findByCustomerIdAndCompetencyGroupId(Long customerId, Long competencyGroupId);

  @EntityGraph(value = "competency.applicability", type = EntityGraph.EntityGraphType.FETCH)
  Competency findByCustomerIdAndId(Long customerId, Long id);

  List<Competency> findByCustomerIdAndCompetencyNameIgnoreCase(Long customerId, String name);

    List<Competency> findByCustomerIdAndStatusNotAndCompetencyNameIgnoreCase(Long customerId,
            CompetencyStatus status, String competencyName);

  List<Competency> findByCustomerIdAndIdIn(Long customerId, List<Long> id);

  List<Competency> findAllByCustomerId(Long customerId);

  Page<Competency> findAllByCustomerIdAndStatusAndCompetencyNameContainingIgnoreCase(Long customerId,
      CompetencyStatus competencyStatus, String competencyName, Pageable pageable);

  List<Competency> findByCustomerIdAndStatusNotAndCompetencyGroupIdAndCompetencyNameIgnoreCase(Long customerId,
      CompetencyStatus status, Long competencyGroupId, String competencyName);

  @Query(value = "SELECT DISTINCT c " + FIND_ALL_COMPETENCIES_POSTFIX_V1,
      countQuery = "SELECT COUNT(DISTINCT c.id) " + FIND_ALL_COMPETENCIES_POSTFIX_V1)
  Page<Competency> findAllByCustomerId(@Param("customerId") Long customerId,
      @Param("competencyIdsFlag") boolean competencyNamesFlag,
      @Param("competencyIds") List<Long> competencyIds,
      @Param("competencyGroupIdsFlag") boolean competencyGroupIdsFlag,
      @Param("competencyGroupId") List<Long> competencyGroupId,
      @Param("jobFunctionIdFlag") boolean jobFunctionIdFlag,
      @Param("jobFunctionId") List<Long> jobFunctionId,
      @Param("payGradeIdsFlag") boolean payGradeIdsFlag,
      @Param("payGradeIds") List<Long> payGradeIds,
      @Param("updatedOnFromFlag") boolean updatedOnFromFlag,
      @Param("updatedOnFrom") Timestamp updatedOnFrom,
      @Param("updatedOnToFlag") boolean updatedOnToFlag,
      @Param("updatedOnTo") Timestamp updatedOnTo, @Param("updatedByFlag") boolean updatedByFlag,
      @Param("updatedBy") List<Long> updatedBy, @Param("statusFlag") boolean statusFlag,
      @Param("statuses") List<CompetencyStatus> statuses,
      Pageable pageable);

  /**
   * method to delete competency based on below params
   *
   * @param customerId
   * @param competencyId
   * @return
   */
  @Modifying
  @Query(
      "update Competency c set c.status = com.peoplehum.competencyManagement.v1.enums"
          + ".CompetencyStatus.INACTIVE, c.isActive = null "
          + "where c.customerId = :customerId and c.id = :competencyId")
  Integer updateStatusByCustomerIdAndId(@Param("customerId") Long customerId,
      @Param("competencyId") Long competencyId);

  /**
   * method to find competencies based on below params
   *
   * @param customerId
   * @param status
   *
   * @return
   */
  List<Competency> findAllByCustomerIdAndStatus(Long customerId, CompetencyStatus status);

  /**
   * method to fetch count by status with customerId
   *
   * @param customerId
   *
   * @return
   */
  @Query("Select C.status as status, count(C.status) as count from Competency C "
      + "where C.customerId = :customerId group by C.status having status != 'INACTIVE'")
  Set<StatusAndCountModelInterface> countByCustomerId(@Param("customerId") Long customerId);

  @Query(value = "SELECT DISTINCT c " + FIND_ALL_APPLICABLE_COMPETENCIES_POSTFIX_V1,
      countQuery = "SELECT COUNT(DISTINCT c.id) " + FIND_ALL_APPLICABLE_COMPETENCIES_POSTFIX_V1)
  Page<Competency> findAllApplicableCompetencies(@Param("customerId") Long customerId,
      @Param("jobFunctionId") Long jobFunctionId, @Param("payGradeId") Long payGradeId,
      @Param("queryValue") String queryValue, @Param("hasQueryValue") Boolean hasQueryValue,
      Pageable pageable);

  interface StatusAndCountModelInterface {
    CompetencyStatus getStatus();
    Long getCount();
  }

  Page<Competency> findAllByCustomerIdAndStatus(Long customerId, CompetencyStatus inactive,
      Pageable pageable);
}
